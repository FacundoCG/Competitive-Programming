while(!bestRectangleToPaint.empty() && points < k){
            pair<ll,ll> bestRectangle = bestRectangleToPaint.top();
            bestRectangleToPaint.pop();
            DBG(bestRectangle);
            DBG(points);
            
            ll width = bestRectangle.first;
            ll height = bestRectangle.second;

            if (width == 1 && height == 1){
                points += 2;
                operations +=1;
            } else if (width < height){
                operations += width;
                bestRectangleToPaint.push({width,height-1});
                points++;
            } else {
                operations += height;
                bestRectangleToPaint.push({width-1,height});
                points++;
            }

            DBG(operations);
            RAYA;
        }

        if (points < k){
            operations = -1;
        }
